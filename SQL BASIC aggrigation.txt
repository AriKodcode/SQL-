2. SELECT status,COUNT(*) as total FROM `orders` GROUP by status
3. SELECT salesRepEmployeeNumber,AVG(creditLimit) as AVG FROM `customers` GROUP by salesRepEmployeeNumber
4. SELECT country,MIN(creditLimit) as MIN_creditlimit,MAX(creditLimit) as MAX_creditlimit FROM `customers` GROUP BY country
5. SELECT officeCode,COUNT(*) as employees FROM `employees` GROUP by officeCode
6. SELECT customerNumber,AVG(amount) as payment FROM `payments` GROUP by customerNumber ORDER BY amount DESC
7. SELECT productVendor,AVG(buyPrice) avg_b_p FROM `products` GROUP BY productVendor
8. SELECT salesRepEmployeeNumber,COUNT(*) as customers FROM `customers` WHERE salesRepEmployeeNumber != 'null' GROUP by salesRepEmployeeNumber
9. SELECT orderDate,COUNT(*) as total FROM `orders` GROUP by orderDate ORDER by total DESC. (2004-11-24)
